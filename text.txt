const hearts = document.getElementById("hearts");
const confetti = document.getElementById("confetti");

const playBtn = document.getElementById("playBtn");
const pauseBtn = document.getElementById("pauseBtn");
const music = document.getElementById("music");

const dateGrid = document.getElementById("dateGrid");
const finalBox = document.getElementById("finalBox");
const finalLine = document.getElementById("finalLine");
const backBtn = document.getElementById("backBtn");

const typeLine = document.getElementById("typeLine");

function typewriter(el, text, speed=34){
  el.innerHTML = "";
  const span = document.createElement("span");
  span.textContent = "";
  const cur = document.createElement("span");
  cur.className = "cursor";
  cur.textContent = "|";
  el.appendChild(span);
  el.appendChild(cur);

  let i = 0;
  (function tick(){
    span.textContent = text.slice(0, i);
    i++;
    if(i <= text.length) setTimeout(tick, speed);
  })();
}
typewriter(typeLine, "Bear is asking you officially üòåüíò");

const heartChars = ["üíó","üíñ","üíò","üíï","‚ù§Ô∏è","üåπ"];
setInterval(() => {
  const s = document.createElement("span");
  s.textContent = heartChars[Math.floor(Math.random()*heartChars.length)];
  s.style.left = (Math.random()*100) + "vw";
  s.style.fontSize = (14 + Math.random()*28) + "px";
  s.style.animationDuration = (4 + Math.random()*4.5) + "s";
  s.style.setProperty("--drift", (Math.random()*120 - 60) + "px");
  hearts.appendChild(s);
  setTimeout(() => s.remove(), 9000);
}, 220);

function launchConfetti(){
  confetti.innerHTML = "";
  confetti.style.display = "block";
  for(let i=0;i<160;i++){
    const c = document.createElement("i");
    c.style.left = (Math.random()*100) + "vw";
    c.style.background = `hsl(${330 + Math.random()*40}, 95%, 60%)`;
    c.style.animationDuration = (2.2 + Math.random()*2.4) + "s";
    c.style.animationDelay = (Math.random()*0.15) + "s";
    confetti.appendChild(c);
  }
  setTimeout(()=> confetti.style.display = "none", 4200);
}

async function tryPlay(){
  try{
    await music.play();
    playBtn.style.display = "none";
    pauseBtn.style.display = "inline-block";
  }catch(e){}
}

playBtn.addEventListener("click", tryPlay);

pauseBtn.addEventListener("click", () => {
  music.pause();
  pauseBtn.style.display = "none";
  playBtn.style.display = "inline-block";
});

yesBtn.addEventListener("click", async () => {
  await tryPlay();
  launchConfetti();
  setTimeout(() => {
    page1.classList.remove("active");
    page2.classList.add("active");
    launchConfetti();
  }, 650);
});

function moveNo(){
  const a = btnArea.getBoundingClientRect();
  const b = noBtn.getBoundingClientRect();
  const maxX = a.width - b.width;
  const maxY = a.height - b.height;
  noBtn.style.left = Math.max(0, Math.floor(Math.random()*maxX)) + "px";
  noBtn.style.top  = Math.max(0, Math.floor(Math.random()*maxY)) + "px";
}
noBtn.addEventListener("mouseenter", moveNo);
noBtn.addEventListener("touchstart", (e)=>{e.preventDefault(); moveNo();},{passive:false});
noBtn.addEventListener("click", moveNo);

dateGrid.addEventListener("click", (e) => {
  const opt = e.target.closest(".opt");
  if(!opt) return;
  finalLine.textContent = `Janae picked: ${opt.dataset.choice}. Bear is planning it üíó`;
  finalBox.style.display = "block";
  launchConfetti();
});

backBtn.addEventListener("click", () => {
  page2.classList.remove("active");
  page1.classList.add("active");
  finalBox.style.display = "none";
});
